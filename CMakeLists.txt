cmake_minimum_required(VERSION 3.20)
project(CraftPacker VERSION 3.0.1 LANGUAGES CXX)

configure_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/CraftPackerVersion.h.in"
    "${CMAKE_CURRENT_BINARY_DIR}/CraftPackerVersion.h"
    @ONLY
)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

# ============================================================
# SECURE BUILD-TIME API KEY INJECTION
# ============================================================
# Reads CF_API_KEY from environment, XOR-encrypts it with a random salt,
# and writes api_keys.h as a generated file.
# If the env var is not set, the key will be empty and CF features will
# gracefully show "API Key Required".
# ============================================================

# Generate random salt for XOR obfuscation
set(CF_SALT "")
set(CF_XOR_KEY "")
set(CF_API_KEY "$ENV{CF_API_KEY}")

if(NOT "${CF_API_KEY}" STREQUAL "")
    # Generate a random 32-byte hex salt using CMake's random
    string(RANDOM LENGTH 64 ALPHABET "0123456789abcdef" CF_SALT)
    # The XOR key will be generated by the Python script
    set(CF_XOR_KEY "${CF_API_KEY}")
    message(STATUS "CF_API_KEY found, will obfuscate at build time")
else()
    message(STATUS "CF_API_KEY not set - CurseForge features will show 'API Key Required'")
endif()

# Find Python for key obfuscation script
find_package(Python3 QUIET COMPONENTS Interpreter)
if(Python3_FOUND AND NOT "${CF_API_KEY}" STREQUAL "")
    # Run the obfuscation script to generate api_keys.h
    configure_file(
        "${CMAKE_CURRENT_SOURCE_DIR}/api_keys.h.in"
        "${CMAKE_CURRENT_BINARY_DIR}/api_keys.h.tmp"
        @ONLY
    )
    add_custom_command(
        OUTPUT "${CMAKE_CURRENT_BINARY_DIR}/api_keys.h"
        COMMAND Python3::Interpreter
        ARGS "${CMAKE_CURRENT_SOURCE_DIR}/scripts/obfuscate_key.py"
            "${CF_API_KEY}"
            "${CF_SALT}"
            "${CMAKE_CURRENT_BINARY_DIR}/api_keys.h"
        DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/scripts/obfuscate_key.py"
        COMMENT "Obfuscating CurseForge API key..."
    )
else()
    # Generate a stub api_keys.h with empty key
    file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/api_keys.h"
"#ifndef API_KEYS_H
#define API_KEYS_H
// Generated at build time - CF_API_KEY was not set
// CurseForge features will be disabled gracefully
#include <array>
#include <cstdint>

namespace ApiKeys {
static constexpr std::array<uint8_t, 0> kCurseForgeKeyEncrypted = {};
static constexpr std::array<uint8_t, 4> kCurseForgeSalt = {0,0,0,0};
inline bool HasCurseForgeKey() { return false; }
}
#endif // API_KEYS_H
"
    )
endif()

# Dependencies: Qt6
find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets Network)
find_package(Qt6 COMPONENTS Sql QUIET)

# Force static linking (MSVC /MT flag) for standalone .exe
if(MSVC)
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
endif()

# For SQLite caching - use bundled or system sqlite
find_package(SQLite3 QUIET)
if(SQLite3_FOUND)
    message(STATUS "Using system SQLite3")
    add_definitions(-DUSE_SYSTEM_SQLITE=1)
else()
    message(STATUS "SQLite3 not found, will use embedded JSON file cache")
    add_definitions(-DUSE_SYSTEM_SQLITE=0)
endif()

# Source files
set(SOURCES
    CraftPacker.cpp
    CraftPacker.h
    resources.qrc
    src/AsyncWorker.cpp
    src/AsyncWorker.h
    src/ModrinthAPI.cpp
    src/ModrinthAPI.h
    src/CurseForgeAPI.cpp
    src/CurseForgeAPI.h
    src/CacheManager.cpp
    src/CacheManager.h
    src/PackExporter.cpp
    src/PackExporter.h
    src/ConflictDetector.cpp
    src/ConflictDetector.h
    src/UpdateChecker.cpp
    src/UpdateChecker.h
    src/miniz.h
    src/miniz_impl.cpp
    src/StringCleaner.h
    src/ThemeManager.h
    src/FolderDebugger.h
    src/FolderDebugger.cpp
    src/DebuggerDashboard.h
    src/DebuggerDashboard.cpp
    src/JarMetadataExtractor.h
    src/JarMetadataExtractor.cpp
)

add_executable(CraftPacker WIN32 ${SOURCES})

# Set the WIN32 executable property
set_property(TARGET CraftPacker PROPERTY WIN32_EXECUTABLE TRUE)

# Include generated api_keys.h directory
target_include_directories(CraftPacker PRIVATE
    ${CMAKE_CURRENT_BINARY_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/src
)

target_link_libraries(CraftPacker PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    Qt6::Network
)

if(Qt6_Sql_FOUND)
    target_link_libraries(CraftPacker PRIVATE Qt6::Sql)
endif()

if(SQLite3_FOUND)
    target_link_libraries(CraftPacker PRIVATE SQLite::SQLite3)
endif()

# Set application icon (Windows)
if(WIN32)
    set(APP_ICON "${CMAKE_CURRENT_SOURCE_DIR}/resources/app.rc")
    if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/resources/app.ico")
        set(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} /NODEFAULTLIB:msvcrt.lib")
        target_sources(CraftPacker PRIVATE ${APP_ICON})
    endif()
endif()

# Install rules
install(TARGETS CraftPacker
    RUNTIME DESTINATION bin
    BUNDLE DESTINATION bin
)
